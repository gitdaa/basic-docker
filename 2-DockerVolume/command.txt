docker build -t volume-app .

# สร้าง named volumes
docker volume create app-data
docker volume create app-logs



docker run -d \
  --name volume-demo \
  -p 3000:3000 \
  -v app-data:/app/data \
  -v app-logs:/app/logs \
  -v ${PWD}/host-config:/app/config \
  -v /app/tmp \
  -e NODE_ENV=production \
  volume-app
